//応用2 関数 unique を宣言してください。 //引数として配列を 1 つとり、重複していない値のみが入った新しい配列を返してください。 /** * @param {Array} array * @returns {Array} 重複していない値のみが入った新しい配列 */ function unique(array){ const result = []; let UnipueBoolean = false; for (const arrayValue of array){ for (let i = 0; i < array.length; i++){ //配列最初から文字が含まれているか確認していく //同じものがあれば booleanはfalseのまま if (arrayValue === array[i]){ UnipueBoolean = false; } else { UnipueBoolean = true; } } //for 終わり if (UnipueBoolean === true){ result.push(arrayValue); } } return result; } test(unique([1, 1, 1, 1, 1]), []); test(unique([1, 3, 4, 5, 2, 4]), [1, 3, 5, 2]);
